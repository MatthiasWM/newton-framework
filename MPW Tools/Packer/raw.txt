progress(const char * inMessage, ...)
{
	if (gDoProgress)
	{
		va_list  ap;
		va_start(ap, inMessage);
		fputs("# ", stderr);
		vfprintf(stderr, inMessage, ap);
		fputs(".\n", stderr);
		va_end(ap);
	}
}


void
v_goof(const char * inLevel, const char * inMessage, va_list ap)
{
	fprintf(stderr,  "### %s - ", gCmdName);
	fprintf(stderr,  "%s - ", inLevel);
	vfprintf(stderr, inMessage, ap);
	fputs(stderr,  ".\n");
}


void
error_message(const char * inMessage, ...)
{
	va_list  ap;
	va_start(ap, inMessage);
	v_goof("error", inMessage, ap);
	va_end(ap);
	exit(2);
}


void
error_message_if(bool inErr, const char * inMessage, ...)
{
	if (inErr) {
		va_list  ap;
		va_start(ap, inMessage);
		v_goof("error", inMessage, ap);
		va_end(ap);
		exit(2);
	}
}


void
warning_message(const char * inMessage, ...)
{
	va_list  ap;
	va_start(ap, inMessage);
	v_goof("warning", inMessage, ap);
	va_end(ap);
}


void
warning_message_if(bool inErr, const char * inMessage, ...)
{
	if (inErr) {
		va_list  ap;
		va_start(ap, inMessage);
		v_goof("warning", inMessage, ap);
		va_end(ap);
	}
}


void
internal_error_if(bool inErr, const char * inMessage, ...)
{
	if (inErr) {
		va_list  ap;
		va_start(ap, inMessage);
		v_goof("internal error", inMessage, ap);
		va_end(ap);
		exit(3);
	}
}


void
usage_error_message(const char ** inUsage, const char * inMessage, ...)
{
	va_list  ap;
	va_start(ap, inMessage);
	v_goof("bad usage", inMessage, ap);
	va_end(ap);

	if (*inUsage != NULL) {
		fprintf(stderr,  "### %s - ", gCmdName);
		while (const char * s = *inUsage++ != NULL) {
			fputs(s, stderr);
		}
	}
	exit(1);
}


void
v_err_goof(const char * inLevel, const char * inMessage, va_list ap)
{
	char buf[256];
	v_goof(inMessage, "error", ap);
	ErrMgrAnon(inLevel, buf);
	fprintf(g-91e, "# %s", buf);	// stderr?
}


bool
ReportOSErr(const short inOSErr, const char * const inMessage, ...)
{
	if (inOSErr != noErr) {
		va_list  ap;
		va_start(ap, inMessage);
		v_err_goof(inOSErr, inMessage, ap);
		va_end(ap);
		return true;
	}
	return false;
}


void
exit_if_OSErr(const short inOSErr, const char * const inMessage, ...)
{
	if (inOSErr != noErr) {
		va_list  ap;
		va_start(ap, inMessage);
		v_err_goof(inOSErr, inMessage, ap);
		va_end(ap);
		exit(2);
	}
}


void *
calloc_or_exit(size_t inNumItems, size_t inSize, const char * inMessage)
{
	void * p = calloc(inNumItems, inSize);
	if (p == NULL) {
		error_message(inMessage, inSize);	// sic
	}
}


void *
malloc_or_exit(size_t inSize, const char * inMessage)
{
	void * p = malloc(inSize);
	if (p == NULL) {
		error_message(inMessage, inSize);
	}
}
